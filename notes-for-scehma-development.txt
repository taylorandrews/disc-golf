Questions:
- Who won the world championship last year?
select player.name, sum(score) as total
from round
join player 
    on round.player_id = player.id
join tournament
    on round.tournament_id = tournament.id
where 
    tournament.name = 'World Championship'
    and tournament.year = strftime('%Y', 'now') - 1
group by player.name
having max(total)
